function getAdvice() {
    var weatherSelected = document.getElementById("weather").value;
    var advice = "";
    
    switch(weatherSelected){
        case "sunny":
            advice = "Wear your sunglasses and sunscreen";
            break;
        case "windy":
            advice = "Wear a jacket and hold onto your hat";
            break;
        case "rainy":
            advice = "Bring an umbrella and wear waterproof shoes";
            break;
        case "snowy":
            advice = "Wear warm clothes and boots";
            break;
        default:
            advice = "Select a weather condition to get advice";
    }
    
    document.getElementById("advice").innerHTML = advice;
}
